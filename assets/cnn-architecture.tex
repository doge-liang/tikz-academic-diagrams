\begin{document}
\begin{tikzpicture}[
    node distance=0.8cm,
    font=\small,
    % Node styles
    conv/.style={draw, rounded corners=3pt, minimum width=2cm, minimum height=0.9cm, fill=blue!15},

    \node[conv] (conv1) {Conv 64@3×3};

    \draw[arrow] (conv3) -- (fc);
